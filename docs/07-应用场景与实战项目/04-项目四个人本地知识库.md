# 个人本地知识库完整实战

## 3. LlamaIndex 基础实战

### 3.1 最简示例

```python
# simple_rag.py
from llama_index.core import VectorStoreIndex, SimpleDirectoryReader
from llama_index.llms.ollama import Ollama
from llama_index.embeddings.ollama import OllamaEmbedding
from llama_index.core import Settings

# 配置 LLM 和 Embedding
Settings.llm = Ollama(model="llama3.1:8b", request_timeout=60.0)
Settings.embed_model = OllamaEmbedding(model_name="nomic-embed-text:latest")

# 1. 加载文档（准备一个 docs 文件夹，放入你的 Markdown/PDF 等）
documents = SimpleDirectoryReader("docs").load_data()

# 2. 创建索引
index = VectorStoreIndex.from_documents(documents)

# 3. 创建查询引擎
query_engine = index.as_query_engine()

# 4. 查询
response = query_engine.query("这篇文档主要讲了什么？")
print(response)
```

### 4.2 构建索引脚本

```python
# build_index.py
import os
from dotenv import load_dotenv
from llama_index.core import VectorStoreIndex, SimpleDirectoryReader, StorageContext, load_index_from_storage
from llama_index.llms.ollama import Ollama
from llama_index.embeddings.ollama import OllamaEmbedding
from llama_index.core import Settings
import chromadb
from llama_index.vector_stores.chroma import ChromaVectorStore

load_dotenv()

# 配置
Settings.llm = Ollama(model="llama3.1:8b", request_timeout=60.0)
Settings.embed_model = OllamaEmbedding(model_name="nomic-embed-text:latest")

PERSIST_DIR = "./storage"
DATA_DIR = "./data"

def build_index():
    """构建索引"""
    print(f"正在从 {DATA_DIR} 加载文档...")
    
    # 加载文档
    documents = SimpleDirectoryReader(DATA_DIR).load_data()
    print(f"加载了 {len(documents)} 个文档块")
    
    # 初始化 Chroma
    db = chromadb.PersistentClient(path="./chroma_db")
    chroma_collection = db.get_or_create_collection("my_knowledge_base")
    vector_store = ChromaVectorStore(chroma_collection=chroma_collection)
    storage_context = StorageContext.from_defaults(vector_store=vector_store)
    
    # 创建索引
    print("正在创建索引...")
    index = VectorStoreIndex.from_documents(
        documents, 
        storage_context=storage_context,
        show_progress=True
    )
    
    # 持久化
    index.storage_context.persist(persist_dir=PERSIST_DIR)
    print(f"索引已保存到 {PERSIST_DIR}")
    return index

def load_index():
    """加载已有索引"""
    if not os.path.exists(PERSIST_DIR):
        return build_index()
    
    print(f"正在从 {PERSIST_DIR} 加载索引...")
    
    # 初始化 Chroma
    db = chromadb.PersistentClient(path="./chroma_db")
    chroma_collection = db.get_or_create_collection("my_knowledge_base")
    vector_store = ChromaVectorStore(chroma_collection=chroma_collection)
    storage_context = StorageContext.from_defaults(
        persist_dir=PERSIST_DIR,
        vector_store=vector_store
    )
    
    index = load_index_from_storage(storage_context)
    return index

if __name__ == "__main__":
    build_index()
```

### 4.3 命令行查询脚本

```python
# query.py
from build_index import load_index
from llama_index.core import Settings
from llama_index.llms.ollama import Ollama
from llama_index.embeddings.ollama import OllamaEmbedding

Settings.llm = Ollama(model="llama3.1:8b", request_timeout=60.0)
Settings.embed_model = OllamaEmbedding(model_name="nomic-embed-text:latest")

def query_knowledge_base(question: str):
    index = load_index()
    query_engine = index.as_query_engine(similarity_top_k=3)
    
    print(f"\n问题: {question}\n")
    response = query_engine.query(question)
    print("回答:", response)
    
    # 显示来源
    print("\n来源文档:")
    for i, node in enumerate(response.source_nodes, 1):
        score = node.score or 0.0
        print(f"{i}. (相似度: {score:.2f})")
        print(node.node.get_content()[:300], "...\n")

if __name__ == "__main__":
    import sys
    if len(sys.argv) > 1:
        question = " ".join(sys.argv[1:])
        query_knowledge_base(question)
    else:
        print("使用: python query.py \"你的问题\"")
```

### 4.4 Streamlit Web 界面

```python
# app.py
import streamlit as st
from build_index import load_index
from llama_index.core import Settings
from llama_index.llms.ollama import Ollama
from llama_index.embeddings.ollama import OllamaEmbedding

# 页面配置
st.set_page_config(page_title="个人知识库", page_icon="📚", layout="wide")

# 配置
Settings.llm = Ollama(model="llama3.1:8b", request_timeout=60.0)
Settings.embed_model = OllamaEmbedding(model_name="nomic-embed-text:latest")

# 侧边栏
st.sidebar.title("⚙️ 设置")
model_option = st.sidebar.selectbox("选择模型", ["llama3.1:8b", "qwen2.5:7b", "mistral:7b"])
top_k = st.sidebar.slider("检索文档数量", 1, 5, 3)
temperature = st.sidebar.slider("Temperature", 0.0, 1.0, 0.7)

# 更新 LLM 设置
Settings.llm = Ollama(model=model_option, temperature=temperature, request_timeout=60.0)

# 主界面
st.title("📚 个人知识库")
st.markdown("---")

# 加载索引
@st.cache_resource
def get_index():
    return load_index()

try:
    index = get_index()
    query_engine = index.as_query_engine(similarity_top_k=top_k)
    
    # 输入区域
    question = st.text_input("🔍 输入你的问题:", placeholder="例如：关于这个项目我需要了解什么？")
    
    if question:
        with st.spinner("正在思考..."):
            response = query_engine.query(question)
            
            # 显示回答
            st.markdown("### 🤖 回答")
            st.info(response.response)
            
            # 显示来源
            with st.expander("📖 查看来源文档"):
                for i, node in enumerate(response.source_nodes, 1):
                    score = node.score or 0.0
                    st.markdown(f"**文档 {i}** (相似度: {score:.2f})")
                    st.text(node.node.get_content()[:500] + "...")

except Exception as e:
    st.error(f"错误: {e}")
    st.info("请先运行 python build_index.py 构建索引！")

# 页脚
st.markdown("---")
st.markdown("💡 提示：把文档放到 data 文件夹，运行 build_index.py 构建索引")
```

### 6.3 混合检索 + 重排序

```python
from llama_index.core import VectorStoreIndex
from llama_index.core.retrievers import VectorIndexRetriever
from llama_index.core.query_engine import RetrieverQueryEngine
from llama_index.core.postprocessor import SimilarityPostprocessor

# 高级查询引擎
index = load_index()
retriever = VectorIndexRetriever(index=index, similarity_top_k=5)

query_engine = RetrieverQueryEngine(
    retriever=retriever,
    node_postprocessors=[
        SimilarityPostprocessor(similarity_cutoff=0.7)  # 过滤低相似度
    ]
)
```

## 8. 常见问题

### Q: 运行太慢怎么办？
A:
- 使用更小的模型（7B 而非 8B/14B）
- 减小 `chunk_size`
- 关闭不必要的后处理

### Q: 如何添加新文档？
A:
1. 将新文档放入 `data/`
2. 重新运行 `python build_index.py`

### Q: 隐私安全吗？
A:
- 完全本地化！
- 数据不上网
- 无需 API Key

## 9. 投喂指南：如何准备文档效果更好？

这一节是**核心中的核心**！文档质量决定了知识库效果！

### 9.1 文档格式优先级（按推荐度排序）

| 格式 | 推荐度 | 原因 |
|------|--------|------|
| **Markdown (.md)** | ⭐⭐⭐⭐⭐ | 结构清晰，带标题层级，完美支持 |
| **纯文本 (.txt)** | ⭐⭐⭐⭐ | 简单，无格式干扰 |
| **PDF (.pdf)** | ⭐⭐⭐ | 支持但可能有排版问题 |
| **Word (.docx)** | ⭐⭐ | 支持，比 PDF 好一点 |
| **网页、PPT 等** | ⭐ | 建议先转成 Markdown 再用 |

**最佳实践**：尽量把你的笔记都转成 Markdown 格式！

---

### 9.2 如何写"好"文档（给 LLM 看）

#### ❌ 糟糕的文档示例
```markdown
# 我的笔记

一些内容...
可能有一些点...
然后又一些其他东西...
这个东西很重要，但我忘记说是什么了...
```

#### ✅ 优秀的文档示例
```markdown
# 2025-06-15 API 鉴权方案

## 问题背景
我们的 API 需要安全认证，防止未授权访问。

## 方案选型
- JWT Token
- OAuth 2.0
- API Key

最终选择：**JWT Token**

## 实现步骤
1. 用户登录成功后，服务器签发 JWT
2. 客户端把 Token 存在 localStorage
3. 每次请求时，把 Token 放在 Header 的 `Authorization` 字段
4. 服务器验证 Token 有效性

## 关键代码
```python
import jwt
from datetime import datetime, timedelta

def create_token(user_id):
    payload = {
        "user_id": user_id,
        "exp": datetime.utcnow() + timedelta(days=7)
    }
    return jwt.encode(payload, SECRET_KEY, algorithm="HS256")
```
```

### 9.4 文档内容最佳实践

| 技巧 | 为什么 | 例子 |
|------|--------|------|
| **标题层级清晰** | 方便分割和检索 | `# 一级`、`## 二级` |

### 9.5 文档分割策略（Chunking）

这是影响 RAG 效果的**最关键参数**之一！

#### 推荐配置（根据文档类型调整）

| 文档类型 | Chunk Size | Chunk Overlap |
|---------|-----------|--------------|
| 技术笔记 | 512-1024 | 100-200 |
| 长文档/论文 | 1024-2048 | 200-300 |
| 代码文件 | 1024-2048 | 150-250 |

#### 如何修改（代码示例）
```python
from llama_index.core.node_parser import SentenceSplitter

# 技术笔记推荐配置
Settings.text_splitter = SentenceSplitter(
    chunk_size=512,      # 每个块 512 token
    chunk_overlap=100    # 重叠 100 token
)
```
